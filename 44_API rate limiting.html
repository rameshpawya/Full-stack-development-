Express Rate Limiting Backend Code


const express = require("express");

const rateLimit =
  require("express-rate-limit");

const app = express();

const limiter = rateLimit({

  windowMs: 1 * 60 * 1000,

  max: 5,

  message:
    "Too many requests, try again later"

});

app.use(limiter);

app.get("/api",(req,res)=>{

  res.json({

    message:"API Request Success"

  });

});

app.listen(5000);
