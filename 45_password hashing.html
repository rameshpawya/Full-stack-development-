Node.js bcrypt Password Hashing


const bcrypt = require("bcrypt");

const password = "mypassword";

const saltRounds = 10;

// Hash Password

bcrypt.hash(
  password,
  saltRounds,
  (err, hash) => {

    console.log(hash);

  }
);

// Compare Password

bcrypt.compare(
  "mypassword",
  hash,
  (err,result) => {

    console.log(result);

  }
);
