const mongoose = require("mongoose");

mongoose.connect(
  "mongodb://127.0.0.1:27017/mydatabase"
)

.then(() => {
  console.log("MongoDB Connected");
})

.catch((error) => {
  console.log(error);
});

const userSchema = new mongoose.Schema({

  name:String,

  email:String

});

const User =
  mongoose.model("User", userSchema);
