Express Centralized Error Handling


const express = require("express");

const app = express();

app.get("/users",(req,res,next)=>{

  try{

    throw new Error(
      "Database Connection Failed"
    );

  }

  catch(error){

    next(error);

  }

});

// Centralized Error Middleware

app.use((error,req,res,next)=>{

  res.status(500).json({

    success:false,

    message:error.message

  });

});

app.listen(5000);
