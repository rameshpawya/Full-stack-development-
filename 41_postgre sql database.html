Node.js PostgreSQL Backend Code


const { Client } = require("pg");

const client = new Client({

  host: "localhost",

  user: "postgres",

  password: "1234",

  database: "company_db",

  port: 5432

});

client.connect()

.then(() => {
  console.log("PostgreSQL Connected");
})

.catch((error) => {
  console.log(error);
});
